 62     var e = $("#rocket-to-top"),

 67     $(window).scroll(function() {
 68         var t = $(document).scrollTop();
 69         t == 0 ? e.css("background-position") == "0px 0px" ? e.fadeOut("slow") : i && (i = !1, $(".level-2").css("opacity", 1), e.delay(100).animate({
 70             marginTop: "-1000px"

 77             }),

 79         })) : e.fadeIn("slow")

 83             opacity: 1
 84         })
 85     },
 86     function() {
 87         $(".level-2").stop(!0).animate({
 88             opacity: 0
 89         })

 92         function t() {
 93             var t = e.css("background-position");
 94             if (e.css("display") == "none" || i == 0) {
 95                 clearInterval(n),
 96                 e.css("background-position", "0px 0px");
 97                 return
 98             }
